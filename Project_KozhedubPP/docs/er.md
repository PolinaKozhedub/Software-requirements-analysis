# ER-диаграмма модели данных ИС

```mermaid
erDiagram
    %% Отношения и связи между сущностями
    Подразделение ||--o{ Сотрудник : "содержит штат"
    Подразделение ||--o{ Бригада : "включает"
    Бригада ||--o{ Сотрудник : "состоит из рабочих"
    Изделие ||--o{ Перемещение : "проходит через"
    Подразделение ||--o{ Перемещение : "принимает на этапе"
    Бригада ||--o{ Перемещение : "выполняет работу"
    Изделие ||--o{ Протокол_испытаний : "проверяется в"
    Лабораторное_оборудование ||--o{ Протокол_испытаний : "используется для"

    %% Описание сущностей и их атрибутов
    Подразделение {
        string Наименование_цеха_или_участка
        string Тип_подразделения
        string ФИО_начальника_мастера_ИТР
    }

    Сотрудник {
        string ФИО
        string Категория_персонала_Рабочий_или_ИТР
        string Должность
    }

    Бригада {
        string Номер_бригады
        string ФИО_бригадира
    }

    Изделие {
        string Серийный_номер
        string Категория_Автобус_или_Сельхозмашина
        int Вместимость_для_автобусов
        string Производительность_для_сельхозмашин
        string Текущий_статус
    }

    Перемещение {
        datetime Дата_и_время_перемещения
        string Шаг_технологического_маршрута
    }

    Протокол_испытаний {
        datetime Дата_проведения_теста
        string Качественные_показатели
        string Результат_Успешно_или_Брак
    }

    Лабораторное_оборудование {
        string Инвентарный_номер_стенда
        string Наименование_оборудования
        string Статус_активности
    }
