# Диаграмма потоков данных (Уровень 1)

```mermaid
graph TB
    %% Внешние сущности
    Dispatcher[Диспетчер цеха]
    HR[Сотрудник ОК]
    Lab[Лаборант]
    Director[Руководитель]

    %% Накопители данных (Хранилища)
    DB_Staff[("База кадров<br>(Сотрудники, бригады, ИТР)")]
    DB_Products[("База изделий<br>(Номенклатура, маршруты)")]
    DB_Tests[("База испытаний<br>(Протоколы, стенды)")]

    %% Процессы
    P1((1. Учет кадров<br>и структур))
    P2((2. Регистрация<br>изделий))
    P3((3. Учет движения<br>по участкам))
    P4((4. Фиксация<br>испытаний))
    P5((5. Сводная<br>аналитика))

    %% Потоки процесса 1: Кадры
    HR -->|"Данные сотрудников и структуры"| P1
    P1 -->|"Запись кадров, ИТР, бригадиров"| DB_Staff
    DB_Staff -->|"Актуальные списки персонала"| P1
    P1 -->|"Статус выполнения операции"| HR

    %% Потоки процесса 2: Регистрация изделий
    Dispatcher -->|"Серийный номер и категория"| P2
    P2 -->|"Запись новой карточки изделия"| DB_Products
    P2 -->|"Подтверждение сохранения"| Dispatcher

    %% Потоки процесса 3: Движение по участкам
    Dispatcher -->|"Данные перемещения (участок, бригада)"| P3
    DB_Staff -->|"Проверка состава бригады и ИТР"| P3
    P3 -->|"Обновление шага маршрута изделия"| DB_Products
    DB_Products -->|"Текущий статус машины"| P3
    P3 -->|"История перемещений"| Dispatcher

    %% Потоки процесса 4: Испытания
    Lab -->|"Данные тестов и оборудования"| P4
    DB_Products -->|"Проверка готовности изделия"| P4
    P4 -->|"Запись протокола испытаний"| DB_Tests
    P4 -->|"Подтверждение регистрации протокола"| Lab

    %% Потоки процесса 5: Аналитика и отчеты
    Director -->|"Параметры отчетов"| P5
    DB_Staff -->|"Выгрузка истории движения кадров"| P5
    DB_Products -->|"Выгрузка объемов выпуска"| P5
    DB_Tests -->|"Выгрузка результатов тестов"| P5
    P5 -->|"Готовые аналитические отчеты"| Director
