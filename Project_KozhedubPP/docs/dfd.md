@startuml
title Диаграмма потоков данных (DFD) — Уровень 1 

top to bottom direction

' Настройка стилей
skinparam Component {
    BackgroundColor<<Актор>> #f9f
    BackgroundColor<<Процесс>> #c2f1ff
    BackgroundColor<<Хранилище>> #fff7c2
    BorderColor #333
    RoundCorner 15
}

' Внешние сущности (Акторы)
component "Сотрудник Отдела кадров (ОК)" as HR <<Актор>>
component "Диспетчер цеха" as Disp <<Актор>>
component "Лаборант" as Lab <<Актор>>
component "Руководитель завода" as Dir <<Актор>>

' Процессы (Функции системы)
component "1. Управление\nкадровой структурой" as P1 <<Процесс>>
component "2. Регистрация\nнового изделия" as P2 <<Процесс>>
component "3. Учет перемещения\nпо участкам" as P3 <<Процесс>>
component "4. Регистрация\nрезультатов тестов" as P4 <<Процесс>>
component "5. Сводная аналитика\nи отчетность" as P5 <<Процесс>>

' Накопители данных (Базы данных / Таблицы)
database "(DB) База кадров и структур" as DB_Staff <<Хранилище>>
database "(DB) База изделий и маршрутов" as DB_Products <<Хранилище>>
database "(DB) База испытаний" as DB_Tests <<Хранилище>>

' Потоки данных для Процесса 1 (Кадры)
HR --> P1 : Данные сотрудников, ИТР, цехов
P1 --> DB_Staff : Запись состава бригад и руководства
DB_Staff --> P1 : Списки доступного персонала
P1 --> HR : Статус выполнения операции

' Потоки данных для Процесса 2 (Регистрация продукции)
Disp --> P2 : Заводской номер, категория машины
P2 --> DB_Products : Новая карточка машины (статус: Сборка)
P2 --> Disp : Подтверждение успешной записи

' Потоки данных для Процесса 3 (Движение по цеху)
Disp --> P3 : ID изделия, целевой участок, № бригады
DB_Staff --> P3 : Данные о квалификации и активности бригады
DB_Products --> P3 : Текущая технологическая позиция
P3 --> DB_Products : Обновление шага маршрута и времени
P3 --> Disp : Визуализация нового трека машины

' Потоки данных для Процесса 4 (Испытания в лаборатории)
Lab --> P4 : ID стенда, параметры тестов, статус
DB_Products --> P4 : Проверка готовности изделия
P4 --> DB_Tests : Запись протокола контроля качества
P4 --> DB_Products : Смена статуса (Готов / Брак)
P4 --> Lab : Форма электронного паспорта для печати

' Потоки данных для Процесса 5 (Аналитика для руководства)
Dir --> P5 : Запрос отчета, фильтры, периоды
DB_Staff --> P5 : Данные о движении штата
DB_Products --> P5 : Данные об объемах выпуска
DB_Tests --> P5 : Статистика брака и прохождения тестов
P5 --> Dir : Сводные таблицы, графики KPI, файлы экспорта
@enduml
